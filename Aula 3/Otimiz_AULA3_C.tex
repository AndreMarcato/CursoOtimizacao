\begin{frame}
	\frametitle{Exemplo Minimização}
	\begin{block}{Utilize o Tableau Simplex para resolver o seguinte problema de programação linear:}
		\begin{equation*}
			\begin{matrix}
				\min Z = x_1 + x_2 - 4x_3 \\
				\text{Sujeito à:} \\
				x_1 + x_2 + 2x_3 \le 9 \\
				x_2 + x_2 - x_3 \le 2 \\
				-x_1 + x_2 + x_3 \le 4 \\
				x_1, x_2, x_3 \ge 0 \\
			\end{matrix}
		\end{equation*}
	\end{block}
